<?php
class Solution {
    function projectionArea($grid) {
        $total = 0;
        foreach($grid[0] as $k => $c){
            $total += max(array_column($grid,$k));
        }
        foreach($grid as $key => $coord){
            $total += max($coord);
            $total += count(array_filter($coord));
        }
        return $total;
    }
}
