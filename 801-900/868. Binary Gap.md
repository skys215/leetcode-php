<?php
class Solution {
    function binaryGap($N) {
        $bin = decbin($N);
        $chars = str_split($bin);
        $len = 0;
        $max = 0;
        $ones = 0;
        foreach($chars as $key=>$char){
            if($char == '1'){
                $ones++;
                $len++;
                if($len>$max){
                    $max = $len;
                }
                $len = 0;
            }
            else{
                $len++;
            }
        }
        return $ones>1?$max:0;
    }
}
