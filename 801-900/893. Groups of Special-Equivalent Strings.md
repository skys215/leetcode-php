<?php
class Solution {
    function numSpecialEquivGroups($A) {
        $words = [];
        foreach($A as $b){
            $odd = $even = [];
            $chars = str_split($b);
            foreach($chars as $key=>$char){
                if($key%2==0){
                    $odd[] = $char;
                }
                else{
                    $even[] = $char;
                }
            }
            sort($odd);
            sort($even);
            $words[implode('',$odd).'/'.implode('',$even)] = true;
        }
        return count($words);
    }
}
