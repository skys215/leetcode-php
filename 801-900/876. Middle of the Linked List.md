<?php
/**
 * Definition for a singly-linked list.
 * class ListNode {
 *     public $val = 0;
 *     public $next = null;
 *     function __construct($val) { $this->val = $val; }
 * }
 */
class Solution {
    function middleNode($head) {
        $items = [$head];
        while($head){
            $items[] = $head;
            $head = $head->next;
        };
        return $items[ceil(count($items)/2)];
    }
}
