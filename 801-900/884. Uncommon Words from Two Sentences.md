<?php
class Solution {
    function uncommonFromSentences($A, $B) {
        $d = array_merge(array_diff(explode(' ', $A), explode(' ', $B)),array_diff(explode(' ', $B), explode(' ', $A)));
        return array_keys(array_filter(array_count_values($d),function($val){
            return $val==1;
        }));
    }
}
