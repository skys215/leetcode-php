<?php
class Solution {

    /**
     * @param String $S
     * @return String
     */
    function toGoatLatin($S) {
        $words = explode(' ', $S);
        $newWords = [];
        foreach($words as $key => $word){
            if(!in_array($word[0],['a','e','i','o','u','A','E','I','O','U'])){
                $word .= $word[0];
                $word = substr($word,1);
            }
            $word .= 'ma'.str_repeat('a', $key+1);
            $newWords[] = $word;
        }
        return implode(' ', $newWords);
    }
}
