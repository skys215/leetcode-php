<?php
class Solution {
    public $max = 0;
    public $level = 0;
    function maxDepth($root) {
        if($root){
            $this->level++;
            if($this->level>=$this->max){
                $this->max = $this->level;
            }
        }
        if($root->children){
            foreach($root->children as $child){
                $this->maxDepth($child);    
            }
        }
        if($root){
            $this->level--;
        }
        return $this->max;
    }
}
