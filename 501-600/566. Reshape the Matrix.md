<?php
class Solution {

    /**
     * @param Integer[][] $nums
     * @param Integer $r
     * @param Integer $c
     * @return Integer[][]
     */
    function matrixReshape($nums, $r, $c) {
        $values = [];
        foreach($nums as $items){
            foreach($items as $item){
                $values[] = $item;
            }
        }
        return count($values)/$c==$r?array_chunk($values, $c):$nums;
    }
}
