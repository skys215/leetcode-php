class Solution {
    public $val = [];
    /**
     * @param Node $root
     * @return Integer[]
     */
    function postorder($root) {
        if(!$root){
            return $this->val;
        }
        foreach($root->children as $child){
            $this->postorder($child);
        }
        $this->val[] = $root->val;
        return $this->val;
    }
}
