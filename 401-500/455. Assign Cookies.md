<?php
class Solution {

    /**
     * @param Integer[] $g
     * @param Integer[] $s
     * @return Integer
     */
    function findContentChildren($g, $s) {
        $total = 0;
        foreach($s as $key => $value){
            while( !is_null(key($g)) ){
                if (current($g)<=$value ){
                    $total++;
                    next($g);
                    break;
                }
                next($g);
            }
        }
        return $total;
    }
}
