<?php
class Solution {

    /**
     * @param String $s
     * @return Integer
     */
    function longestPalindrome($s) {
        $s = str_split($s);
        $amounts = array_count_values($s);
        $odd = [];
        $even = [];
        array_walk($amounts, function($v, $k) use (&$odd, &$even){
            if($v%2 != 0){
                $odd[$k] = $v;
            }
            else{
                $even[$k] = $v;
            }
        });
        $odds = count($odd);
        if(!empty($odds)){
            $odds--;
        }
        $total = array_sum($even) + array_sum($odd) - $odds;
        return $total;
    }
}
