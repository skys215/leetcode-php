class Solution {

    /**
     * @param Node $root
     * @return Integer[][]
     */
    public $level = 0;
    public $values = [];
    function levelOrder($root) {
        if(is_null($root)){
            return $this->values;
        }        
        if(!isset($this->values[$this->level])){
            $this->values[$this->level] = [];
        }
        $this->values[$this->level][] = $root->val;
        foreach($root->children as $child){
            $this->level++;
            $this->levelOrder($child);
            $this->level--;
        }
        
        return $this->values;
    }
}
