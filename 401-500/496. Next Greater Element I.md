<?php
class Solution {

    /**
     * @param Integer[] $nums1
     * @param Integer[] $nums2
     * @return Integer[]
     */
    function nextGreaterElement($nums1, $nums2) {
        $result = [];
        foreach($nums1 as $key => $value){
            $greater = -1;
            $start = false;
            for($i = 0; $i<count($nums2); $i++){
                if($nums2[$i] == $value){
                    $start = true;
                }
                if($start && $nums2[$i]>$value){
                    $greater = $nums2[$i];
                    break;
                }
            }
            $result[] = $greater;
        }
        return $result;
    }
}
