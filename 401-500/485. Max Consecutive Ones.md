<?php
class Solution {

    /**
     * @param Integer[] $nums
     * @return Integer
     */
    function findMaxConsecutiveOnes($nums) {
        $max = 0;
        $current = 0;
        foreach($nums as $val){
            if($val == 1){
                $current++;
                if($current>$max){
                    $max = $current;
                }
            }
            else{
                $current = 0;
            }
        }
        return $max;
    }
}
