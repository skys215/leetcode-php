<?php
class Solution {
    function fizzBuzz($n) {
        $fb = [];
        $a = range(1,$n);
        foreach($a as $val){
            if($val%3==0&&$val%5==0){
                $fb[] = 'FizzBuzz';
                continue;
            }
            if($val%3==0){
                $fb[] = 'Fizz';
                continue;
            } 
            if($val%5==0){
                $fb[] = 'Buzz';
                continue;
            }
            $fb[] = (string)$val;
        };
        return $fb;
    }
}
