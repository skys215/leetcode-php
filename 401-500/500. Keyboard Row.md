<?php
class Solution {
    function findWords($words) {
        return array_filter($words, function($val){
            $val = strtolower($val);
            $q = ['q','w','e','r','t','y','u','i','o','p'];
            $a = ['a','s','d','f','g','h','j','k','l'];
            $z = ['z','x','c','v','b','n','m'];
            
            return !(array_diff(array_unique(str_split($val)),$q) && array_diff(array_unique(str_split($val)),$a) &&
                array_diff(array_unique(str_split($val)),$z));
        });
    }
}
