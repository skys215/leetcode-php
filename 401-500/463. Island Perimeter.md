<?php
class Solution {
    function islandPerimeter($grid) {
        $edges = 0;
        $width = count($grid[0]);
        $height = count($grid);
        foreach($grid as $y=>$row){
            foreach($row as $x=>$col){
                if($col==0){
                    continue;
                }
                if(!isset($grid[$y][$x+1])||$grid[$y][$x+1]==0){
                    $edges++;
                }
                if(!isset($grid[$y][$x-1])||$grid[$y][$x-1]==0){
                    $edges++;
                }
                if(!isset($grid[$y+1][$x])||$grid[$y+1][$x]==0){
                    $edges++;
                }
                if(!isset($grid[$y-1][$x])||$grid[$y-1][$x]==0){
                    $edges++;
                }
            }
        }
        return $edges;
    }
}
