<?php
/**
 * Definition for a binary tree node.
 * class TreeNode {
 *     public $val = null;
 *     public $left = null;
 *     public $right = null;
 *     function __construct($value) { $this->val = $value; }
 * }
 */
class Solution {
    protected $str = '';

    /**
     * @param TreeNode $t
     * @return String
     */
    function tree2str($t) {
        $this->str .= $t->val;
        if(!is_null($t->left)){
            $this->str .= '(';
            $this->tree2str($t->left);
            $this->str .= ')';
        }
        if(!is_null($t->right)){
            if(is_null($t->left)){
                $this->str .= '()';
            }
            $this->str .= '(';
            $this->tree2str($t->right);
            $this->str .= ')';
        }
        return $this->str;
    }
}
