<?php
/**
 * Definition for a binary tree node.
 * class TreeNode {
 *     public $val = null;
 *     public $left = null;
 *     public $right = null;
 *     function __construct($value) { $this->val = $value; }
 * }
 */
class Solution {
    function searchBST($root, $val) {
        if($root->val == $val){
            return $root;
        }
        $a = NULL;
        $b = NULL;
        if($root->left){
            $a = $this->searchBST($root->left, $val);
            if($a){
                return $a;
            }
        }
        if($root->right){
            $b = $this->searchBST($root->right, $val);
            if($b){
                return $b;
            }
        }
        return NULL;
    }
}
