<?php
class Solution {

    /**
     * @param String $A
     * @param String $B
     * @return Boolean
     */
    function rotateString($aa, $bb) {
        $A = str_split($aa);
        $B = str_split($bb);
        $blen = count($B);
        $i = 0;
        while(($blen != $i) && ($A != $B)){
            array_unshift($B, array_pop($B));
            $i++;
        }
        return $A == $B;
    }
}
