<?php
class Solution {
    function selfDividingNumbers($left, $right) {
        $nums = range($left,$right+1);
        $result = [];
        foreach($nums as $num){
            if($num<10){
                $result[] = $num;
                continue;
            }
            if($num%10 == 0){
                continue;
            }
            $ds = array_unique(str_split($num));
            $stopFlag = false;
            foreach($ds as $d){
                if($d == 0){
                    $stopFlag = true;
                    break;
                }
                if($num%$d!=0){
                    $stopFlag = true;
                    break;
                }
            }
            if(!$stopFlag){
                $result[] = $num;
            }
        }
        return $result;
    }
}
