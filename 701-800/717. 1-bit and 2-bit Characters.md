<?php
class Solution {

    /**
     * @param Integer[] $bits
     * @return Boolean
     */
    function isOneBitCharacter($bits) {
        $stack = [];
        $isOneBit = false;
        foreach($bits as $bit){
            if(isset($stack[0])){
                array_pop($stack);
                $isOneBit = false;
            }
            else{
                if($bit == 1){
                    $stack[] = $bit;
                }
                else{
                    $isOneBit = true;
                }
            }
        }
        
        return $isOneBit;
    }
}
