<?php
/**
 * Definition for a singly-linked list.
 * class ListNode {
 *     public $val = 0;
 *     public $next = null;
 *     function __construct($val = 0, $next = null) {
 *         $this->val = $val;
 *         $this->next = $next;
 *     }
 * }
 */
class Solution {

    /**
     * @param ListNode $head
     * @param Integer $val
     * @return ListNode
     */
    function removeElements($head, $val) {
        $current = $head;
        while(!is_null($current)){
            if($current->val === $val){
                $head = $current->next;
            }
            if($current->next->val === $val){
                $current->next = $current->next->next;
            }
            else{
                $current = $current->next;
            }
        }
        return $head;
    }
}
