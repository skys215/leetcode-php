<?php
class Solution {

    /**
     * @param String $s
     * @param String $t
     * @return Boolean
     */
    function isAnagram($s, $t) {
        $arrs = str_split($s);
        $arrt = str_split($t);
        sort($arrs);
        sort($arrt);
        return implode($arrs) === implode($arrt);
    }
}
