<?php
/**
 * Definition for a singly-linked list.
 * class ListNode {
 *     public $val = 0;
 *     public $next = null;
 *     function __construct($val) { $this->val = $val; }
 * }
 */
class Solution {
    /**
     * @param ListNode $head
     * @return ListNode
     */
    function reverseList($head) {
        $stack = [];
        while(true){
            $stack[] = $head;
            $head = $head->next;
            if(is_null($head)){
                break;
            }
        }
        $root = $head = array_pop($stack);
        while($stack){
            $head->next = array_pop($stack);
            $head = $head->next;
        }
        $head->next = null;
        return $root;
    }
}
