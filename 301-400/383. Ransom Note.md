<?php
class Solution {

    /**
     * @param String $ransomNote
     * @param String $magazine
     * @return Boolean
     */
    function canConstruct($ransomNote, $magazine) {
        if(!$ransomNote){
            return true;
        }
        $rArr = array_count_values(str_split($ransomNote));
        $mArr = array_count_values(str_split($magazine));
        foreach($rArr as $char => $amount){
            if(!isset($mArr[$char])){
                return false;
            }
            if($mArr[$char]<$amount){
                return false;
            }
        }
        return true;
    }
}
