<?php
class Solution {

    /**
     * @param String $s
     * @param String $t
     * @return String
     */
    function findTheDifference($s, $t) {
        $ss = array_count_values(str_split($s));
        $tt = array_count_values(str_split($t));
        $diff = array_diff_key($tt, $ss) + array_diff($tt, $ss) + array_diff_assoc($tt, $ss);
        return key($diff);
    }
}
