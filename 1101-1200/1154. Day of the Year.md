<?php
class Solution {

    /**
     * @param String $date
     * @return Integer
     */
    public $daysOfMonth = [
        31,28,31,30,31,30,
        31,31,30,31,30,31,
    ];
    function dayOfYear($date) {
        list($year, $month, $day) = explode('-', $date);
        $year = intval($year);
        $month = intval($month);
        $day = intval($day);
        $isLeap = intval(($month>2) && ($year%4 == 0) && ($year%100!=0));

        return array_sum(array_slice($this->daysOfMonth,0,$month-1)) + $day + $isLeap;
    }
}
