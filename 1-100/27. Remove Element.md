<?php
class Solution {

    /**
     * @param Integer[] $nums
     * @param Integer $val
     * @return Integer
     */
    function removeElement(&$nums, $val) {
        $remain = 0;
        foreach($nums as $key => $num){
            if($num == $val){
                unset($nums[$key]);
            }
            $remain++;
        }
        return $remain;
    }
}
