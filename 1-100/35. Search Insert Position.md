<?php
class Solution {

    /**
     * @param Integer[] $nums
     * @param Integer $target
     * @return Integer
     */
    function searchInsert($nums, $target) {
        foreach($nums as $index => $num){
            if($num == $target){
                return $index;
            }
            if($num > $target){
                return max(0,$index);
            }
        }
        return $index+1;
    }
}
