<?php
class Solution {

    /**
     * @param String $s
     * @return Integer
     */
    function lengthOfLastWord($s) {
        $len = 0;
        if($s == ''){
            return $len;
        }
        $str = str_split($s);
        $prev = NULL;
        foreach($str as $char){
            if($char != ' '){
                if($prev == ' '){
                    $len = 0;
                }
                $len++;
            }
            $prev = $char;
        }
        return $len;
    }
}
