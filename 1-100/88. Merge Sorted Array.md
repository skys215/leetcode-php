<?php
class Solution {

    /**
     * @param Integer[] $nums1
     * @param Integer $m
     * @param Integer[] $nums2
     * @param Integer $n
     * @return NULL
     */
    function merge(&$nums1, $m, $nums2, $n) {
        $index = 0;
        for($i = 0; $i<$m+$n; $i++){
            if($index>=$n){
                break;
            }
            if($nums1[$i]>=$nums2[$index]){
                //å¾€åç§»
                for($j = $m+$index; $j>$i; $j--){
                    $nums1[$j] = $nums1[$j-1];
                }
                $nums1[$i] = $nums2[$index];
                $index++;
            }
            if($i>=$m+$index){
                $nums1[$i] = $nums2[$index];
                $index++;
            }
        }
        return $nums1;
    }
}
