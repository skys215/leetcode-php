<?php
class Solution {

    /**
     * @param Integer $n
     * @return Integer
     */
    function climbStairs($n) {
        $current = 1;
        $prev = 0;
        $pprev = 0;
        for($i=0; $i<$n; $i++){
            $pprev = $prev;
            $prev = $current;
            $current = $pprev + $prev;
        }
        return $current;
    }
}
