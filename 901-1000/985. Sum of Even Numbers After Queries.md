<?php
class Solution {

    /**
     * @param Integer[] $A
     * @param Integer[][] $queries
     * @return Integer[]
     */
    function sumEvenAfterQueries($A, $queries) {
        $evens = array_filter($A,function($val){
            return $val%2==0;
        });
        $total = array_sum($evens);
        $sums = [];
        foreach($queries as $query){
            if($A[$query[1]]%2 == 0){
                $total -= $A[$query[1]];
            }
            if(($A[$query[1]]+$query[0])%2==0){         
                $total += $query[0] + $A[$query[1]];   
            }
            $sums[] = $total;
            $A[$query[1]] += $query[0];
        }
        return $sums;
    }
}
