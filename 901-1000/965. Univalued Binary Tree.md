class Solution {
    private $vals = [];
    function isUnivalTree($root) {
        $this->getVal($root);
        return count(array_count_values(array_filter($this->val,function($v){
            return !is_null($v);
        })))==1;
    }
    
    function getVal($root){
        $this->val[] = $root->val;
        if($root->left){
            $this->getVal($root->left);
        }
        if($root->right){
            $this->getVal($root->right);
        }
    }
}
