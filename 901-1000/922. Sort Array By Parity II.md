<?php
pubiclass Solution {
    function sortArrayByParityII($A) {
        $odd = array_filter($A,function($val){
            return ($val&1); 
        });
        $odd = array_values($odd);
        $even = array_filter($A, function($val){
            return (!($val&1)); 
        });
        $even = array_values($even);
        $a = [];
        foreach($odd as $key => $o){
            $a[] = $even[$key];
            $a[] = $o;
        }
        return $a;
    }
}
