<?php
class Solution {

    /**
     * @param Integer[] $A
     * @return Integer
     */
    function largestPerimeter($A) {
        rsort($A);
        $length = count($A);
        for($i = 0; $i<$length-2; $i++){
            if(  ($A[$i]   + $A[$i+1] > $A[$i+2])
              && ($A[$i]   + $A[$i+2] > $A[$i+1])
              && ($A[$i+1] + $A[$i+2] > $A[$i])
              ){
                return $A[$i] + $A[$i+1] + $A[$i+2];
            }
        }
        return 0;
    }
}
