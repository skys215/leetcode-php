<?php
class Solution {

    /**
     * @param String[] $logs
     * @return String[]
     */
    function reorderLogFiles($logs) {
        $letters = [];
        $digits = [];
        foreach($logs as $log){
            $values = explode(' ',$log);
            $id = array_shift($values);
            $firstLetter = ord($values[0][0]) - ord('0');
            if($firstLetter>=0 && $firstLetter<=9){
                $digits[] = $log;
            }
            else{
                $letters[implode(' ',$values)] = $log;
            }
        }
        ksort($letters);
        return $letters+$digits;
    }
}
