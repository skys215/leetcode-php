<?php
/**
 * Definition for a binary tree node.
 * class TreeNode {
 *     public $val = null;
 *     public $left = null;
 *     public $right = null;
 *     function __construct($val = 0, $left = null, $right = null) {
 *         $this->val = $val;
 *         $this->left = $left;
 *         $this->right = $right;
 *     }
 * }
 */
class Solution {
    /**
     * @param TreeNode $root
     * @return Boolean
     */
    function isBalanced($root) {
        $left = 0;
        $right = 0;
        if(!is_null($root->left)){
            $left = $this->isBalanced($root->left);
            if($left === false){
                return false;
            }
        }
        if(!is_null($root->right)){
            $right = $this->isBalanced($root->right);
            if($right === false){
                return false;
            }
        }
        if(abs($left-$right) > 1){
            return false;
        }
        return max($left, $right)+1;
    }
    
}
