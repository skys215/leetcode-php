<?php
/**
 * Definition for a binary tree node.
 * class TreeNode {
 *     public $val = null;
 *     public $left = null;
 *     public $right = null;
 *     function __construct($value) { $this->val = $value; }
 * }
 */
class Solution {
    public $max = 0;
    public $level = 0;
    /**
     * @param TreeNode $root
     * @return Integer
     */
    function maxDepth($root) {
        if($root){
            $this->level++;
        }
        if($this->level>=$this->max){
            $this->max = $this->level;
        }
        if($root->left){
            $this->maxDepth($root->left);
        }
        if($root->right){
            $this->maxDepth($root->right);
        }
        $this->level--;
        return $this->max;
    }
}
