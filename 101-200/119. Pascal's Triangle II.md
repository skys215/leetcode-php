<?php
class Solution {

    /**
     * @param Integer $rowIndex
     * @return Integer[]
     */
    function getRow($rowIndex) {
        $current = [];
        if($rowIndex === 0){
            return [1];
        }
        if($rowIndex ===1 ){
            return [1,1];
        }
        $prev = $this->getRow($rowIndex-1);
        $current = range(0,$rowIndex);
        $current[0] = 1;
        $current[$rowIndex] = 1;
        $mid = ceil($rowIndex/2);
        for($i=1; $i<=$mid; $i++){
            $left = $prev[$i-1];
            $right = $prev[$i];
            $current[$i] = $current[$rowIndex-$i] = $left + $right;
        }
        return $current;
    }
}
