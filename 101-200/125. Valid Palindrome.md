<?php
class Solution {

    /**
     * @param String $s
     * @return Boolean
     */
    function isPalindrome($s) {
        $len = strlen($s);
        $mid = $len/2;
        for($i=0, $j=$len-1; $i<$j;){
            //var_dump($i.'-'.$j);
            $left = $s[$i];
            $right = $s[$j];
            if(!$this->isASCII($left)){
                $i++;
                continue;
            }
            if(!$this->isASCII($right)){
                $j--;
                continue;
            }
            if(strtolower($left) != strtolower($right)){
                return false;
            }
            $i++;
            $j--;
        }
        return true;
    }

    function isASCII($char){
        $ord = ord($char);
        return ($ord >=ord('0')
               && $ord<=ord('9'))
               || ($ord>=ord('A')
               && $ord<=ord('Z'))
               ||( $ord>=ord('a')
               && $ord<=ord('z'));
    }
}
