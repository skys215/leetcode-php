<?php
class Solution {

    /**
     * @param Integer $n
     * @return String
     */
    function convertToTitle($n) {
        $string = '';

        while($n!=0){
            $remain = ($n-1)%26;
            $string = chr(ord('A') + $remain) . $string;
            $n = (int)floor(($n-$remain)/26);
        }
        return $string;
    }
}
